29 /**
30  * Implementation of the V2 catalog API for Vertica.
31  *
32  * This is not currently a full catalog implementation. This simply returns a table object upon request,
33  * and indicates whether the table already exists. This is in place to allow for the Ignore and ErrorIfExist
34  * save modes.
35  */
36 class VerticaDatasourceV2Catalog() extends TableCatalog{
37   var readSetupInterface: DSConfigSetupInterface[ReadConfig] = new DSReadConfigSetup
38 
39   override def initialize(name: String, options: CaseInsensitiveStringMap): Unit = {
40     VerticaDatasourceV2Catalog.catalogOptions = Some(options)
41   }
42 
43   override def tableExists(ident: Identifier): Boolean = {
44     val opt = VerticaDatasourceV2Catalog.getOptions.getOrElse(throw new NoSuchTableException(ident))
45 
46     val table = new VerticaTable(opt, readSetupInterface)
47     val schema = table.schema()
48 
49     schema.nonEmpty
50   }
51 
52   override def name: String = "VerticaCatalog"
53 
54   @throws[NoSuchNamespaceException]
55   override def listTables(namespace: Array[String]): Array[Identifier] = throw new NoCatalogException
56 
57   @throws[NoSuchTableException]
58   override def loadTable(ident: Identifier): Table = {
59     val opt = VerticaDatasourceV2Catalog.getOptions.getOrElse(throw new NoSuchTableException(ident))
60 
61     new VerticaTable(opt, readSetupInterface)
62   }
63 
64   @throws[TableAlreadyExistsException]
65   @throws[NoSuchNamespaceException]
66   def createTable(ident: Identifier, schema: StructType, partitions: Array[Transform], properties: java.util.Map[String, String]): Table = {
67     loadTable(ident)
68   }
69 
70 
71   @throws[NoSuchTableException]
72   def alterTable(ident: Identifier, changes: TableChange*): Table = throw new NoCatalogException
73 
74   def dropTable(ident: Identifier): Boolean = throw new NoCatalogException
75 
76 
77   @throws[NoSuchTableException]
78   @throws[TableAlreadyExistsException]
79   def renameTable(oldIdent: Identifier, newIdent: Identifier): Unit = throw new NoCatalogException
80 }
