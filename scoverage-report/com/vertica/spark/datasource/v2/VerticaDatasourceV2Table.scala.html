32 /**
33  * Represents a Vertica table to Spark.
34  *
35  * Supports Read and Write functionality.
36  */
37 class VerticaTable(caseInsensitiveStringMap: CaseInsensitiveStringMap, readSetupInterface: DSConfigSetupInterface[ReadConfig] = new DSReadConfigSetup) extends Table with SupportsRead with SupportsWrite {
38 
39   // Cache the scan builder so we don't build it twice
40   var scanBuilder : Option[VerticaScanBuilder] = None
41 
42   /**
43    * A name to differentiate this table from other tables
44    *
45    * @return A string representing a unique name for the table.
46    */
47   override def name(): String = "VerticaTable" // TODO: change this to db.tablename
48 
49 
50   /**
51    * Should reach out to SQL layer and return schema of the table.
52    *
53    * @return Spark struct type representing a table schema.
54    */
55   override def schema(): StructType = {
56     // Check if there's a valid read config with schema for the table, if not return empty schema
57     readSetupInterface.validateAndGetConfig(caseInsensitiveStringMap.asScala.toMap) match {
58       case Invalid(_) => new StructType()
59       case Valid(_) => this.newScanBuilder(caseInsensitiveStringMap).build().readSchema()
60     }
61 
62   }
63 
64   /**
65    * Returns a list of capabilities that the table supports.
66    *
67    * @return Set of [[TableCapability]] representing the functions this source supports.
68    */
69   override def capabilities(): util.Set[TableCapability] =
70     Set(TableCapability.BATCH_READ, TableCapability.BATCH_WRITE, TableCapability.OVERWRITE_BY_FILTER,
71       TableCapability.TRUNCATE, TableCapability.ACCEPT_ANY_SCHEMA).asJava
72 
73   /**
74    * Returns a scan builder for reading from Vertica
75    *
76    * @return [[v2.VerticaScanBuilder]]
77    */
78   override def newScanBuilder(options: CaseInsensitiveStringMap): ScanBuilder =
79   {
80     val logger = LogProvider.getLogger(classOf[VerticaTable])
81     this.scanBuilder match {
82       case Some(builder) => builder
83       case None =>
84         val config = readSetupInterface.validateAndGetConfig(options.asScala.toMap) match {
85           case Invalid(errList) =>
86             ErrorHandling.logAndThrowError(logger, ErrorList(errList.toNonEmptyList))
87           case Valid(cfg) => cfg
88         }
89         logger.debug("Config loaded")
90 
91         // Pushdown of aggregates added in spark 3.2, detect spark version so we return class with right feature support
92         var sparkNewerThan31 = true
93         try {
94           val sparkVersion = SparkSession.getActiveSession.get.version
95           val versionList = sparkVersion.split("\\.").map(_.toInt)
96           if(versionList.length >= 2 && versionList(0) == 3 && versionList(1) < 2) {
97             sparkNewerThan31 = false
98           }
99         }
100         catch { // Couldn't recgonize version string, assume newer version
101           case _: java.lang.NumberFormatException => sparkNewerThan31 = true
102         }
103 
104         val scanBuilder = if(sparkNewerThan31) {
105           classOf[VerticaScanBuilderWithPushdown]
106             .getDeclaredConstructor(classOf[ReadConfig], classOf[DSConfigSetupInterface[ReadConfig]])
107             .newInstance(config, readSetupInterface)
108         }
109         else {
110           classOf[VerticaScanBuilder]
111             .getDeclaredConstructor(classOf[ReadConfig], classOf[DSConfigSetupInterface[ReadConfig]])
112             .newInstance(config, readSetupInterface)
113         }
114         this.scanBuilder = Some(scanBuilder)
115         scanBuilder
116     }
117   }
118 
119   /**
120    * Returns a write builder for writing to Vertica
121    *
122    * @return [[v2.VerticaWriteBuilder]]
123    */
124   def newWriteBuilder(info: LogicalWriteInfo): WriteBuilder =
125   {
126     new VerticaWriteBuilder(info, new DSWriteConfigSetup(schema = Some(info.schema)))
127   }
128 }
