21 /**
22   * Interface responsible for writing to the Vertica source.
23   *
24   * This interface is initiated and called from each spark worker.
25   */
26 trait DSWriterInterface {
27   /**
28     * Called before reading to perform any needed setup with the given configuration.
29     */
30   def openWrite(): ConnectorResult[Unit]
31 
32   /**
33     * Called to write an individual row to the datasource.
34     */
35   def writeRow(row: InternalRow): ConnectorResult[Unit]
36 
37   /**
38     * Called from the executor to cleanup the individual write operation
39     */
40   def closeWrite(): ConnectorResult[Unit]
41 
42   /**
43     * Called by the driver to commit all the write results
44     */
45   def commitRows(): ConnectorResult[Unit]
46 }
47 
48 /**
49  * Writer class, agnostic to the kind of pipe used for the operation (which VerticaPipe is used)
50  *
51  * @param config Configuration data definining the write operation.
52  * @param uniqueId Unique identifier for this specific writer. The writer for each partition should have a different ID.
53  * @param pipeFactory Factory returning the underlying implementation of a pipe between us and Vertica, to use for write.
54  */
55 class DSWriter(config: WriteConfig, uniqueId: String, pipeFactory: VerticaPipeFactoryInterface = VerticaPipeFactory) extends DSWriterInterface {
56 
57   private val pipe = pipeFactory.getWritePipe(config)
58   private var blockSize = 0L
59 
60   private var data = List[InternalRow]()
61 
62   def openWrite(): ConnectorResult[Unit] = {
63     for {
64       size <- pipe.getDataBlockSize
65       _ <- pipe.startPartitionWrite(uniqueId)
66       _ = this.blockSize = size
67     } yield ()
68   }
69 
70   def writeRow(row: InternalRow): ConnectorResult[Unit] = {
71     data = data :+ row
72     if(data.length >= blockSize) {
73       pipe.writeData(DataBlock(data)) match {
74         case Right(_) =>
75           data = List[InternalRow]()
76           Right(())
77         case Left(errors) => Left(errors)
78       }
79     }
80     else {
81       Right(())
82     }
83   }
84 
85   def closeWrite(): ConnectorResult[Unit] = {
86     if(data.nonEmpty) {
87        val ret = pipe.writeData(DataBlock(data))
88        pipe.endPartitionWrite()
89        ret
90     }
91     else {
92       pipe.endPartitionWrite()
93     }
94   }
95 
96   def commitRows(): ConnectorResult[Unit] = {
97     val ret = pipe.commit()
98     // Ensure all connections are closed, including read connections used by the write operation
99     val _ = pipeFactory.closeJdbcLayers()
100     ret
101   }
102 }
