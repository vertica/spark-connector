23 /**
24   * Interface responsible for reading from the Vertica source.
25   *
26   * This class is initiated and called from each spark worker.
27   */
28 trait DSReaderInterface {
29 
30   /**
31    * Starts the read operation
32    */
33   def openRead(): ConnectorResult[Unit]
34 
35   /**
36    * Called by spark to read an individual row
37    *
38    * @return The next row if it exists, if the read is done None will be returned.
39    */
40   def readRow(): ConnectorResult[Option[InternalRow]]
41 
42   /**
43    * Called when all reading is done, to perform any needed cleanup operations.
44    */
45   def closeRead(): ConnectorResult[Unit]
46 }
47 
48 
49 /**
50  * Reader class, agnostic to the kind of pipe used for the operation (which VerticaPipe is used)
51  *
52  * @param config Configuration data for the operation.
53  * @param partition Information representing what this reader needs to read. Will be a portion of the overall read operation.
54  * @param pipeFactory Factory returning the underlying implementation of a pipe between us and Vertica, to use for read.
55  */
56 class DSReader(config: ReadConfig, partition: InputPartition, pipeFactory: VerticaPipeFactoryInterface = VerticaPipeFactory) extends DSReaderInterface {
57   private val pipe = pipeFactory.getReadPipe(config)
58 
59   private var block: Option[DataBlock] = None
60   private var i: Int = 0
61 
62   def openRead(): ConnectorResult[Unit] = {
63     i = 0
64     partition match {
65       case verticaPartition: VerticaPartition => pipe.startPartitionRead(verticaPartition)
66       case _ => Left(InvalidPartition().context(
67           "Unexpected state: partition of type 'VerticaPartition' was expected but not received"))
68     }
69   }
70 
71   def readRow(): ConnectorResult[Option[InternalRow]] = {
72     val ret = for {
73       // Get current or next data block
74       dataBlock <- block match {
75         case None =>
76           pipe.readData match {
77             case Left(err) => Left(err)
78             case Right(data) =>
79               block = Some(data)
80               Right(data)
81           }
82         case Some(block) => Right(block)
83       }
84 
85       // Get row from block. If this is the last row of the block, reset the block
86       _ <- if (dataBlock.data.isEmpty) {
87         Left(DoneReading())
88       } else {
89         Right(())
90       }
91 
92       row = dataBlock.data(i)
93       _ = i += 1
94       _ = if (i >= dataBlock.data.size) {
95         block = None
96         i = 0
97       } else {
98         ()
99       }
100     } yield Some(row)
101 
102     ret match {
103       case Right(optRow) => Right(optRow)
104       case Left(err) => err match {
105         case DoneReading() => Right(None) // If there's no more data to be read, return nothing for row
106         case _ => Left(err) // If something else went wrong, return error
107       }
108     }
109   }
110 
111   def closeRead(): ConnectorResult[Unit] = {
112     pipe.endPartitionRead()
113   }
114 }
