28 /**
29  * Entry-Point for Spark V2 Datasource.

35 class VerticaSource extends TableProvider with SupportsCatalogOptions {
36 
37   /**
38    * Used for read operation to get the schema for the table being read from
39    *
40    * @param caseInsensitiveStringMap A string map of options that was passed in by user to datasource
41    * @return The table's schema in spark StructType format
42    */
43   override def inferSchema(caseInsensitiveStringMap: CaseInsensitiveStringMap): StructType = {
44     val table = getTable(schema = StructType(Nil), partitioning = Array.empty[Transform], properties = caseInsensitiveStringMap)
45     table.schema()
46   }
47 
48   /**
49    * Gets the structure representing a Vertica table
50    *
51    * @param schema       StructType representing table schema, used for write
52    * @param partitioning specified partitioning for the table
53    * @param properties   A string map of options that was passed in by user to datasource
54    * @return [[VerticaTable]]
55    */
56   override def getTable(schema: StructType,
57                         partitioning: Array[Transform],
58                         properties: util.Map[String, String]): Table = {
59     new VerticaTable(new CaseInsensitiveStringMap(properties))
60   }
61 
62 
63   override def extractIdentifier(options: CaseInsensitiveStringMap): Identifier = {
64     val name = options.asScala.toMap.getOrElse("table", "")
65     Identifier.of(Array[String](), name)
66   }
67 
68   private val CATALOG_NAME = "vertica"
69   override def extractCatalog(options: CaseInsensitiveStringMap): String = {
70     // Add all passed in options to spark catalog options
71     VerticaDatasourceV2Catalog.setOptions(options)
72 
73     // Set the spark conf for catalog class
74     SparkSession.getActiveSession match {
75       case Some(session) => session.conf.set("spark.sql.catalog." + CATALOG_NAME, "com.vertica.spark.datasource.v2.VerticaDatasourceV2Catalog")
76       case None => throw new ConnectorException(MissingSparkSessionError())
77     }
78 
79     CATALOG_NAME
80   }
