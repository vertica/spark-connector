19 /**
20  * Utils class for translating Spark pushdown filters into Vertica-compatible SQL where clauses.
21  */
22 object PushdownUtils {
23   private def wrapText(value: Any): String = {
24     value match {
25       case d: java.lang.Number =>
26         if (d.doubleValue.isInfinite) {
27           "'" + value.toString + "'"
28         } else {
29           value.toString
30         }
31       case _ => "'" + value.toString + "'"
32     }
33   }
34 
35   /**
36    * Takes a filter, and if we support it, turns it into a format we can add to our queries, therefore pushing down said filter.
37    *
38    * @param filter Spark-defined filter class.
39    * @return A PushFilter containing a string to add to a select from Vertica, or a NonPushFilter if we don't support that filter.
40    */
41   def genFilter(filter: Filter): Either[NonPushFilter, PushFilter] = {
42     filter match {
43       case EqualTo(attribute, value) => Right(PushFilter(filter,
44         "(" + "\"" + attribute + "\"" + " = " + wrapText(value) + ")"))
45       case GreaterThan(attribute, value) => Right(PushFilter(filter,
46         "(" + "\"" + attribute + "\"" + " > " + wrapText(value) + ")"))
47       case GreaterThanOrEqual(attribute, value) => Right(PushFilter(filter,
48         "(" + "\"" + attribute + "\"" + " >= " + wrapText(value) + ")"))
49       case LessThan(attribute, value) => Right(PushFilter(filter,
50         "(" + "\"" + attribute + "\"" + " < " + wrapText(value) + ")"))
51       case LessThanOrEqual(attribute, value) => Right(PushFilter(filter,
52         "(" + "\"" + attribute + "\"" + " <= " + wrapText(value) + ")"))
53       case In(attribute, value) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" + " in " +
54         "(" + value.map(x => wrapText(x)).mkString(", ") + ")" + ")"))
55       case IsNull(attribute) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" + " is NULL" + ")"))
56       case IsNotNull(attribute) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" + " is NOT NULL" + ")"))
57       case And(left, right) => (for {
58         pushFilterLeft <- genFilter(left)
59         pushFilterRight <- genFilter(right)
60       } yield PushFilter(filter, "(" + pushFilterLeft.filterString + " AND " + pushFilterRight.filterString + ")"))
61         .left.map(_ => NonPushFilter(filter))
62       case Or(left, right) => (for {
63         pushFilterLeft <- genFilter(left)
64         pushFilterRight <- genFilter(right)
65       } yield PushFilter(filter, "(" + pushFilterLeft.filterString + " OR " + pushFilterRight.filterString + ")"))
66         .left.map(_ => NonPushFilter(filter))
67       case Not(child) => genFilter(child) match {
68         case Left(_) => Left(NonPushFilter(filter))
69         case Right(pushFilter) => Right(PushFilter(filter, "(" + " NOT " + pushFilter.filterString + ")"))
70       }
71       case StringStartsWith(attribute, value) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" +
72         " like " + wrapText(value + "%") + ")"))
73       case StringEndsWith(attribute, value) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" +
74         " like " + wrapText("%" + value) + ")"))
75       case StringContains(attribute, value) => Right(PushFilter(filter, "(" + "\"" + attribute + "\"" +
76         " like " + wrapText("%" + value + "%") + ")"))
77       case _ => Left(NonPushFilter(filter))
78     }
79   }
80 }
