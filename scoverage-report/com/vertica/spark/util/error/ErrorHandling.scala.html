26 trait ConnectorError {
27   // Adds context to an error
28   def context(text: String): ConnectorError = ContextError(text, this)
29 
30   // Gets the context for debugging. This context is more helpful for developers.
31   def getFullContext: String
32 
33   // Get the underlying error object. This can be used to determine what kind of error occurred.
34   def getUnderlyingError: ConnectorError = this
35 
36   // Gets a user-friendly error message.
37   def getUserMessage: String = this.getFullContext
38 }

51 object ErrorHandling {
52   type Result[+E, +T] = Either[E, T]
53 
54   type ConnectorResult[T] = Result[ConnectorError, T]
55   type SchemaResult[T] = Result[SchemaError, T]
56   type JdbcResult[T] = Result[JdbcError, T]
57 
58   def addCause(errorText: String, throwable: Throwable): String = {
59     errorText +
60     "\n\nCaused by:\n" +
61     throwable.toString +
62     "\nStack trace:\n" +
63     throwable.getStackTrace.mkString("\n")
64   }
65 
66   def addUserFriendlyCause(errorText: String, throwable: Throwable): String = {
67     errorText + "\nCaused by: " + throwable.getMessage
68   }
69 
70   def appendErrors(errorText1: String, errorText2: String): String = {
71     errorText1 + "\n" + errorText2
72   }
73 
74   def logAndThrowError(logger: Logger, error: ConnectorError): Nothing = {
75     logger.error(error.getFullContext)
76     throw new ConnectorException(error)
77   }
78 
79   val invariantViolation: String = "This is likely a bug and should be reported to the developers here:\n" +
80     "https://github.com/vertica/spark-connector/issues"
81 }
