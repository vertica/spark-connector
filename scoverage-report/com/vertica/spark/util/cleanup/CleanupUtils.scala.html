23 /**
24  * Structure containing cleanup information for a given portion of a file.
25  *
26  * @param filename The file to check for cleanup.
27  * @param fileIdx Which portion of the file is done being read.
28  * @param fileRangeCount How many portions of the file exist.
29  */
30 final case class FileCleanupInfo(filename: String, fileIdx: Int, fileRangeCount: Int)
31 
32 /**
33  * Interface for cleanup util handling cleaning up files being accessed in a distributed fashion.
34  *
35  */
36 trait CleanupUtilsInterface {
37 
38   /**
39    * Mark that this portion of the file is done with and cleanup if necessary.
40    *
41    * The cleanup step will only happen after this is called for each portion of the file.
42    * @param fileStoreLayer Interface to interact with the filestore where files requiring cleaning are.
43    * @param fileCleanupInfo Cleanup information for a portion of the file.
44    */
45   def checkAndCleanup(fileStoreLayer: FileStoreLayerInterface, fileCleanupInfo: FileCleanupInfo) : ConnectorResult[Unit]
46 
47   /**
48    * Cleanup all files
49    *
50    * @param fileStoreLayer Interface to interact with the filestore where files requiring cleaning are.
51    * @param path Path of directory
52    */
53   def cleanupAll(fileStoreLayer: FileStoreLayerInterface, path: String) : ConnectorResult[Unit]
54 }
55 
56 class CleanupUtils extends CleanupUtilsInterface {
57   private val logger = LogProvider.getLogger(classOf[CleanupUtils])
58   private def recordFileName(filename: String, idx: Int) = filename + ".cleanup" + idx
59 
60   def cleanupAll(fileStoreLayer: FileStoreLayerInterface, path: String) : ConnectorResult[Unit] = {
61     // Cleanup parent dir (unique id)
62     val p = new Path(s"$path")
63     val parent = p.getParent
64     if(parent != null) {
65       fileStoreLayer.removeDir(parent.toString)
66       Right(())
67     }
68     else {
69       Left(CleanupError(path))
70     }
71   }
72 
73   private def cleanupParentDirIfEmpty(fileStoreLayer: FileStoreLayerInterface, path: String): Either[ConnectorError, Unit]= {
74     for {
75       parentPath <- getParentHadoopPath(path)
76       allFiles <- fileStoreLayer.getFileList(parentPath)
77       parentPath2 <- getParentHadoopPath(parentPath)
78       _ <- if(allFiles.isEmpty) fileStoreLayer.removeDir(parentPath2) else Right(())
79     } yield ()
80   }
81 
82   /**
83    * Uses java-based hadoop path to retrieve the parent path. Checks for null safety here.
84    */
85   private def getParentHadoopPath(path: String): ConnectorResult[String] = {
86     val p = new Path(s"$path")
87     val parent = p.getParent
88     if(parent != null) Right(parent.toString) else Left(ParentDirMissingError(path))
89   }
90 
91   private def performCleanup(fileStoreLayer: FileStoreLayerInterface, fileCleanupInfo: FileCleanupInfo ): ConnectorResult[Unit] = {
92     val filename = fileCleanupInfo.filename
93 
94     for {
95       // Delete all portions
96       _ <- (0 until fileCleanupInfo.fileRangeCount).map(idx => {
97         logger.debug("Removing: " + recordFileName(filename, idx))
98         fileStoreLayer.removeFile(recordFileName(filename, idx))}).toList.sequence
99 
100       // Delete the original file
101       _ = logger.debug("Removing parquet: " + filename)
102       _ <- fileStoreLayer.removeFile(filename)
103 
104       // Delete the directory if empty
105       _ <- this.cleanupParentDirIfEmpty(fileStoreLayer, filename)
106     } yield ()
107   }
108 
109   override def checkAndCleanup(fileStoreLayer: FileStoreLayerInterface, fileCleanupInfo: FileCleanupInfo): ConnectorResult[Unit] = {
110     val filename = fileCleanupInfo.filename
111     logger.info("Doing partition cleanup of file: " + filename)
112 
113     for {
114       // Create the file for this portion
115       _ <- fileStoreLayer.createFile(recordFileName(filename, fileCleanupInfo.fileIdx))
116 
117       _ = logger.debug("File: " + filename + ", Checking file existance")
118       _ = logger.debug("File: " + filename + ", File idx: " + fileCleanupInfo.fileIdx)
119       _ = logger.debug("File: " + filename + ", File range count: " + fileCleanupInfo.fileRangeCount)
120 
121       // Check if all portions are written
122       filesExist <- (0 until fileCleanupInfo.fileRangeCount).map(idx => {
123            logger.debug("Checking existence: " + recordFileName(filename, idx))
124            fileStoreLayer.fileExists(recordFileName(filename, idx))
125         }).toList.sequence
126       allExist <- Right(filesExist.forall(x => x))
127 
128       _ = logger.debug("File: " + filename + ", filesExist: " + filesExist.toString())
129       _ = logger.debug("File: " + filename + ", All exist: " + allExist)
130 
131       _ <- if(allExist) {
132         logger.debug("File: " + filename + ", Performing cleanup")
133         performCleanup(fileStoreLayer, fileCleanupInfo)
134       } else {
135         Right(())
136       }
137 
138     } yield ()
139   }
