34 /**
35  * Column list, validated on the type level to escape unquoted semicolons (SQL Injection prevention)
36  */
37 class ValidColumnList private (value: String) extends Serializable {
38   override def toString: String = value
39 }
40 
41 object ValidColumnList {
42   private def checkStringForUnquotedSemicolon(str: String): Boolean = {
43     var i = 0
44     var inQuote = false
45     var isUnquotedSemi = false
46     for(c <- str) {
47       if(c == '"') inQuote = !inQuote
48       if(c == ';' && !inQuote) isUnquotedSemi = true
49       i += 1
50     }
51 
52     isUnquotedSemi
53   }
54 
55   final def apply(value: String): ValidationResult[Option[ValidColumnList]] = {
56     if (!checkStringForUnquotedSemicolon(value)) {
57       Some(new ValidColumnList(value)).validNec
58     } else {
59       UnquotedSemiInColumns().invalidNec
60     }
61   }
62 }
