57 /**
58  * File permissions class validating it's either in octal format (ie 777) or user-group-other format,
59  * ie rwxrwxrwx
60  */
61 class ValidFilePermissions private (value: String) extends Serializable {
62   override def toString: String = value
63 }
64 
65 object ValidFilePermissions {
66 
67   // octal format, ie 777 or 0750
68   private def isValidOctal(str: String): Boolean = {
69     str.length <= 4 && str.forall(_.isDigit)
70   }
71 
72   // user-group-other format
73   private def isValidLetterPerms(str: String): Boolean = {
74     str.forall(c => "crwxdt-".contains(c))
75   }
76 
77   final def apply(value: String): ValidationResult[ValidFilePermissions] = {
78     val validOctal = isValidOctal(value)
79     val validLetterParams = isValidLetterPerms(value)
80     if (validOctal || validLetterParams) {
81       new ValidFilePermissions(value).validNec
82     } else {
83       InvalidFilePermissions().invalidNec
84     }
85   }
86 }
