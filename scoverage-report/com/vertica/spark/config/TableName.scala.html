18 object EscapeUtils {
19   def sqlEscape(str: String, char: Char = '\"'): String = {
20     val c = char.toString
21     str.replace(c, c + c)
22   }
23 
24   def sqlEscapeAndQuote(str: String): String = {
25     "\"" + sqlEscape(str) + "\""
26   }
27 }

41 /**
42  * Represents a fully qualified tablename in Vertica.
43  *
44  * @param name Name of the table
45  * @param dbschema Optionally, the schema of the table. Public schema will be assumed if not specified.
46  */
47 final case class TableName(name: String, dbschema: Option[String]) extends TableSource {
48 
49   /**
50    * Returns the full name of the table, escaped and surrounded with double quotes to prevent injection
51    * and allow for special characters.
52    */
53   def getFullTableName : String = {
54     dbschema match {
55       case None => EscapeUtils.sqlEscapeAndQuote(name)
56       case Some(schema) => EscapeUtils.sqlEscapeAndQuote(schema) + "." + EscapeUtils.sqlEscapeAndQuote(name)
57     }
58   }
59 
60   /**
61    * The table's name is used as an identifier for the operation.
62    */
63   override def identifier: String = name
64 }
