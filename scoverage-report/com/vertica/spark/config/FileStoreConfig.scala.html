49 /**
50  * Represents configuration for a filestore used by the connector.
51  *
52  * There is not currently much user configuration for the filestore beyond the address to connect to.
53  * @param baseAddress Address to use in the intermediate filesystem
54  * @param sessionId Unique id for a given connector operation
55  * @param preventCleanup A boolean that prevents cleanup if specified to true
56  * @param AWSOptions Options that specify AWS credentials for using S3 storage
57  */
58 final case class FileStoreConfig(baseAddress: String, sessionId: String, preventCleanup: Boolean, awsOptions: AWSOptions) {
59   val defaultFS =
60     if(baseAddress.startsWith("/")) {
61       SparkSession.getActiveSession match {
62         case Some(session) =>
63           val hadoopConf = session.sparkContext.hadoopConfiguration
64           val filepath = Option(hadoopConf.get("fs.defaultFS")) match {
65             case Some(path) => Right(path)
66             case None => Left(HDFSConfigError().context("No fs.defaultFS value supplied in HDFS config"))
67           }
68           filepath
69         case None => Left(NoSparkSessionFound())
70       }
71     }
72     else {
73       None
74     }
75 
76   val newBaseAddress = defaultFS match {
77     case Right(prefix) => prefix + baseAddress
78     case _ => baseAddress
79   }
80 
81   def address: String = {
82     val delimiter = if(baseAddress.takeRight(1) == "/" || baseAddress.takeRight(1) == "\\") "" else "/"
83     // Create unique directory for session
84     baseAddress.stripSuffix(delimiter) + delimiter + sessionId
85     newBaseAddress.stripSuffix(delimiter) + delimiter + sessionId
86   }
87 
88   def externalTableAddress: String = {
89     val delimiter = if(baseAddress.takeRight(1) == "/" || baseAddress.takeRight(1) == "\\") "" else "/"
90     // URL for directory without session ID
91     baseAddress.stripSuffix(delimiter) + delimiter
92     newBaseAddress.stripSuffix(delimiter) + delimiter
93   }
94 }
